#!/usr/bin/env bash
set -euo pipefail

# setup-server.sh
# Prepares a Debian/Ubuntu server for deploying a Docker-based hosting control panel.
# Usage: sudo ./setup-server.sh [--user NAME] [--db postgres|mariadb|none] [--swap MB] [--no-monitoring]
#
# Defaults:
#   user = whms
#   db = postgres
#   swap = 2048 (MB)
#   no-monitoring = false (will not install grafana/prometheus by default unless you omit flag)
#
# The script:
# - updates apt, installs core utils
# - installs Docker Engine + compose plugin
# - installs Node.js LTS via NodeSource
# - installs Git, build-essential
# - installs Postgres or MariaDB if requested
# - creates a system user and adds to 'docker' group
# - sets up UFW (allow ssh/http/https)
# - creates /opt/whms directories and a sample docker-compose infra (traefik + db)
# - optional swapfile creation
# - prints next steps at the end

### CONFIG / ARGS
USER_NAME="whms"
DB_CHOICE="postgres"   # postgres | mariadb | none
SWAP_MB=2048
INSTALL_MONITORING=true

# parse flags
while [[ ${#} -gt 0 ]]; do
  case "$1" in
    --user) USER_NAME="$2"; shift 2 ;;
    --db) DB_CHOICE="$2"; shift 2 ;;
    --swap) SWAP_MB="$2"; shift 2 ;;
    --no-monitoring) INSTALL_MONITORING=false; shift 1 ;;
    --help|-h) echo "Usage: sudo ./setup-server.sh [--user NAME] [--db postgres|mariadb|none] [--swap MB] [--no-monitoring]"; exit 0 ;;
    *) echo "Unknown arg: $1"; echo "Use --help"; exit 1 ;;
  esac
done

echo "==> Starting server bootstrap"
echo "User: $USER_NAME, DB: $DB_CHOICE, swap: ${SWAP_MB}MB, monitoring: ${INSTALL_MONITORING}"

# basic checks
if [[ $(id -u) -ne 0 ]]; then
  echo "This script must be run as root. Use sudo."
  exit 1
fi

# detect distro
if [ -f /etc/os-release ]; then
  . /etc/os-release
  DISTRO_ID="${ID_LIKE:-$ID}"
else
  echo "Cannot detect OS. Exiting."
  exit 1
fi

if ! echo "$DISTRO_ID" | grep -qiE "debian|ubuntu"; then
  echo "This script is written for Debian/Ubuntu family (cloud VPS). Detected: $DISTRO_ID"
  echo "You may adapt it for other distros."
  exit 1
fi

export DEBIAN_FRONTEND=noninteractive

### 1) apt update + core packages
echo "-> Updating apt and installing core packages..."
apt-get update -y
apt-get upgrade -y
apt-get install -y \
  apt-transport-https ca-certificates curl gnupg lsb-release software-properties-common \
  git build-essential vim unzip jq

### 2) create system user (non-login is optional)
if id "$USER_NAME" >/dev/null 2>&1; then
  echo "User $USER_NAME already exists, skipping creation."
else
  echo "-> Creating user '$USER_NAME'..."
  useradd -m -s /bin/bash "$USER_NAME"
  mkdir -p /home/"$USER_NAME"/.ssh
  chown -R "$USER_NAME":"$USER_NAME" /home/"$USER_NAME"/.ssh
fi

### 3) install Docker (official repo) and docker-compose plugin
if command -v docker >/dev/null 2>&1; then
  echo "Docker already installed: $(docker --version || true)"
else
  echo "-> Installing Docker Engine..."
  curl -fsSL https://download.docker.com/linux/$(. /etc/os-release && echo "$ID")/gpg | gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
  echo \
    "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/$(. /etc/os-release && echo "$ID") \
    $(lsb_release -cs) stable" | tee /etc/apt/sources.list.d/docker.list > /dev/null
  apt-get update -y
  apt-get install -y docker-ce docker-ce-cli containerd.io docker-compose-plugin

  # enable and start docker
  systemctl enable --now docker
fi

echo "-> Ensuring $USER_NAME is in docker group..."
usermod -aG docker "$USER_NAME" || true

### 4) install Node.js LTS (NodeSource)
if command -v node >/dev/null 2>&1; then
  echo "Node.js already installed: $(node --version || true)"
else
  echo "-> Installing Node.js LTS (NodeSource)..."
  curl -fsSL https://deb.nodesource.com/setup_lts.x | bash -
  apt-get install -y nodejs
fi

### 5) install database (Postgres or MariaDB) if requested
if [[ "$DB_CHOICE" == "postgres" ]]; then
  if command -v psql >/dev/null 2>&1; then
    echo "Postgres already installed: $(psql --version || true)"
  else
    echo "-> Installing PostgreSQL..."
    apt-get install -y postgresql postgresql-contrib
    systemctl enable --now postgresql
    echo "Postgres installed."
  fi
elif [[ "$DB_CHOICE" == "mariadb" ]]; then
  if command -v mysql >/dev/null 2>&1; then
    echo "MariaDB/MySQL already installed: $(mysql --version || true)"
  else
    echo "-> Installing MariaDB server..."
    apt-get install -y mariadb-server
    systemctl enable --now mariadb
    echo "MariaDB installed. Run 'sudo mysql_secure_installation' to secure."
  fi
else
  echo "-> Skipping DB installation as requested."
fi

### 6) optional: create swapfile if requested and not present
if [[ "$SWAP_MB" -gt 0 ]]; then
  if swapon --show | grep -q ''; then
    echo "-> Checking if swap exists..."
    if swapon --show | awk 'NR>0 {print $1}' | grep -q ''; then
      echo "Swap already enabled. Skipping swap creation."
    else
      echo "-> Creating ${SWAP_MB}MB swapfile..."
      fallocate -l "${SWAP_MB}M" /swapfile || dd if=/dev/zero of=/swapfile bs=1M count="$SWAP_MB"
      chmod 600 /swapfile
      mkswap /swapfile
      swapon /swapfile
      echo '/swapfile none swap sw 0 0' >> /etc/fstab
    fi
  else
    echo "-> Creating ${SWAP_MB}MB swapfile..."
    fallocate -l "${SWAP_MB}M" /swapfile || dd if=/dev/zero of=/swapfile bs=1M count="$SWAP_MB"
    chmod 600 /swapfile
    mkswap /swapfile
    swapon /swapfile
    echo '/swapfile none swap sw 0 0' >> /etc/fstab
  fi
else
  echo "Swap creation skipped (swap size 0)."
fi

### 7) install certbot for Let's Encrypt (optional)
if ! command -v certbot >/dev/null 2>&1; then
  echo "-> Installing certbot (for manual cert management if needed)..."
  apt-get install -y certbot
fi

### 8) setup UFW firewall (allow ssh/http/https) — safe defaults
if command -v ufw >/dev/null 2>&1; then
  echo "-> Configuring UFW firewall (allow 22, 80, 443)..."
  ufw allow OpenSSH || true
  ufw allow 80/tcp || true
  ufw allow 443/tcp || true
  ufw --force enable || true
else
  apt-get install -y ufw
  ufw allow OpenSSH
  ufw allow 80/tcp
  ufw allow 443/tcp
  ufw --force enable
fi

### 9) Create base directories and set ownership
WHMS_ROOT="/opt/whms"
SITES_DIR="${WHMS_ROOT}/sites"
LETSENCRYPT_DIR="${WHMS_ROOT}/letsencrypt"
REGISTRY_DIR="${WHMS_ROOT}/registry"
LOGS_DIR="${WHMS_ROOT}/logs"

echo "-> Creating base directories under ${WHMS_ROOT}..."
mkdir -p "$SITES_DIR" "$LETSENCRYPT_DIR" "$REGISTRY_DIR" "$LOGS_DIR"
chown -R "$USER_NAME":"$USER_NAME" "$WHMS_ROOT"
chmod -R 750 "$WHMS_ROOT"

### 10) Create a sample docker-compose infra (traefik + postgres) for quick start
COMPOSE_DIR="${WHMS_ROOT}/infra"
mkdir -p "$COMPOSE_DIR"
cat > "${COMPOSE_DIR}/docker-compose.yml" <<'EOF'
version: "3.8"
services:
  traefik:
    image: traefik:v2.10
    command:
      - "--providers.docker=true"
      - "--providers.docker.exposedbydefault=false"
      - "--entryPoints.web.address=:80"
      - "--entryPoints.websecure.address=:443"
      - "--api.insecure=false"
      - "--certificatesResolvers.le.httpChallenge.entryPoint=web"
      - "--certificatesResolvers.le.acme.httpChallenge=true"
      - "--certificatesResolvers.le.acme.email=admin@example.com"
      - "--certificatesResolvers.le.acme.storage=/letsencrypt/acme.json"
    ports:
      - "80:80"
      - "443:443"
      - "8080:8080" # Grafana/trafik dashboard port mapping (optional)
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - ../letsencrypt:/letsencrypt
    networks:
      - infra

  postgres:
    image: postgres:15
    environment:
      POSTGRES_USER: whms
      POSTGRES_PASSWORD: change_this_password
      POSTGRES_DB: whmsdb
    volumes:
      - postgres-data:/var/lib/postgresql/data
    networks:
      - infra

volumes:
  postgres-data:

networks:
  infra:
    driver: bridge
EOF

chown -R "$USER_NAME":"$USER_NAME" "$COMPOSE_DIR"

### 11) Optional monitoring stack (Prometheus + Grafana) using docker compose
if [[ "$INSTALL_MONITORING" == true ]]; then
  MON_DIR="${WHMS_ROOT}/monitoring"
  mkdir -p "$MON_DIR"
  cat > "${MON_DIR}/docker-compose.yml" <<'EOF'
version: "3.8"
services:
  grafana:
    image: grafana/grafana:latest
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_USER=admin
      - GF_SECURITY_ADMIN_PASSWORD=admin
    volumes:
      - grafana-data:/var/lib/grafana
    networks:
      - infra
  prometheus:
    image: prom/prometheus:latest
    ports:
      - "9090:9090"
    volumes:
      - prometheus-data:/prometheus
    networks:
      - infra
volumes:
  grafana-data:
  prometheus-data:
networks:
  infra:
    external: false
EOF
  chown -R "$USER_NAME":"$USER_NAME" "$MON_DIR"
  echo "Monitoring compose created at ${MON_DIR}/docker-compose.yml (not started automatically)."
else
  echo "Monitoring stack skipped by request."
fi

### 12) Pull a test nginx image to ensure docker can pull
echo "-> Pulling a small test image (nginx:alpine) to validate Docker pull..."
su - "$USER_NAME" -c "docker pull nginx:alpine" || true

### 13) Enable basic system services and login message
cat > /etc/motd <<EOF
WHMS server bootstrap completed.

Directories:
  - ${WHMS_ROOT}
  - ${SITES_DIR}
  - ${COMPOSE_DIR}

Next steps (as ${USER_NAME} or root):
  - Edit ${COMPOSE_DIR}/docker-compose.yml (change email, passwords)
  - cd ${COMPOSE_DIR} && docker compose up -d
  - Add your Node backend under /opt/whms/backend (git clone ...)
  - Configure Traefik and DNS (use a wildcard A record or point domains)
  - If using Postgres: sudo -u postgres psql to create additional DB/users
EOF

echo "-> Finished bootstrap. Summary:"
echo "  - WHMS root: $WHMS_ROOT (owner: $USER_NAME)"
echo "  - Sample infra compose: ${COMPOSE_DIR}/docker-compose.yml"
if [[ "$INSTALL_MONITORING" == true ]]; then
  echo "  - Monitoring compose: ${MON_DIR}/docker-compose.yml"
fi
echo "  - To start infra: cd ${COMPOSE_DIR} && docker compose up -d"
echo "  - To run monitoring (optional): cd ${MON_DIR} && docker compose up -d"
echo
echo "Important: edit passwords / email addresses in the docker-compose files before going to production."
echo
echo "Bootstrap complete ✅"
